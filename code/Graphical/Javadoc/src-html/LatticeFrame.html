002    /*
003     *  An implementation of the q-state Potts model
004     *  Uses a square lattice with periodic boundary conditions.
005     *  Metropolis & Wolff Dynamic Algorithm implemented.
006     *  NB temp is in units of Kelvin^2/Joule and J = kB = 1
007     *
008     *  Richard Hanes               17/12/04
009     */
010    import java.io.*;
011    import java.awt.*;
012    import java.awt.event.*;

014    /**
015     *  Contains all the information and methods to manipulate a Lattice
016     *
017     * @author    Richard Hanes
018     */
019    class Lattice {
020            boolean running = true;
021            boolean metropolis = false;
022            boolean sizeChanged = false, statesChanged = false, resetMe = false;
023            int size, states, systemEnergy, newSize, newStates;
024            int[] magnetisation;
025            /**
026             *  The 2d array which holds the current state of each point in the lattice
027             */
028            public int latticeArray[][];
029            boolean[][] cluster;
030    
031            double exponentials[];
032            double temp, tempGradient, addProb;
033    
034    
035            /**
036             *Constructor for the Lattice object
037             *
038             * @param  size          Linear Size of the Lattice
039             * @param  states        No of Spin states in the simulation
040             * @param  temp          Initial temperature of the Lattice
041             * @param  tempGradient  Initial Heating/Cooling Gradient
042             */
043            public Lattice(int size, int states, double temp, double tempGradient) {
044                    this.size = size;
045                    this.states = states;
046                    this.temp = temp;
047                    this.tempGradient = tempGradient;
048                    this.latticeArray = new int[size][size];
049                    //Used to store Metropolis transition Probabilities
050                    this.exponentials = new double[4];
051                    // Stores number of points in each state
052                    this.magnetisation = new int[states];
053                    // Wolff cluster add probability
054                    this.addProb = (1.0 - Math.exp(-1.0 / temp));
055                    this.cluster = new boolean[size][size];
056                    resetSimulation();
057            }
058    
059    
060            /**
061             *  Re-Read Lattice Parameters and Reset the lattice
062             */
063            public void resetSimulation() {
064                    resetMagnetArray();
065                    if (metropolis == false) {
066                            resetClusterArray();
067                    }
068                    calculateExponentials();
069                    addProb = (1.0 - Math.exp(-1.0 / temp));
070                    randomiseLattice();
071                    calculateSystemMagnetisation();
072                    systemEnergy = calculateTotalSystemEnergy();
073            }
074    
075    
076            /**
077             *  Reset the Boolean Cluster Array (Wolff Algorithm)
078             */
079            public void resetClusterArray() {
080                    for (int i = 0; i < size; i++) {
081                            for (int j = 0; j < size; j++) {
082                                    cluster[i][j] = false;
083                            }
084                    }
085            }
086    
087    
088            /**
089             *  Reset the Magnetisation Array, where the number of points in each state is stored
090             */
091            public void resetMagnetArray() {
092                    for (int i = 0; i < states; i++) {
093                            magnetisation[i] = 0;
094                    }
095            }
096    
097    
098            /**
099             *  Count the number of points in each state
100             */
101            public void calculateSystemMagnetisation() {
102                    int currentPointState;
103    
104                    for (int i = 0; i < size; i++) {
105                            for (int j = 0; j < size; j++) {
106                                    currentPointState = latticeArray[i][j];
107                                    if (currentPointState <= states) {
108                                            magnetisation[currentPointState - 1]++;
109                                    }
110                            }
111                    }
112            }
113    
114    
115            /**
116             *  Calculate the flip probabilities (Metropolis Algorithm)
117             */
118            public void calculateExponentials() {
119                    for (int i = 1; i <= 4; i++) {
120                            exponentials[i - 1] = Math.exp(-i / temp);
121                    }
122            }
123    
124    
125            /**
126             *  Set each point on the lattice to a random state (Hot Start)
127             */
128            public void randomiseLattice() {
129                    for (int i = 0; i < size; i++) {
130                            for (int j = 0; j < size; j++) {
131                                    // Load a random state into each lattice point
132                                    this.latticeArray[i][j] = (int) (Math.random() * states) + 1;
133                            }
134                    }
135            }
136    
137    
138            /**
139             *  Go through the entire lattice and calculate the energy from each point
140             *
141             * @return    Total System Energy
142             */
143            public int calculateTotalSystemEnergy() {
144                    int energy = 0;
145                    for (int i = 0; i < size; i++) {
146                            for (int j = 0; j < size; j++) {
147                                    energy += calculateEnergy(i, j);
148                            }
149                    }
150                    return energy;
151            }
152    
153    
154            /**
155             *  Calculate the Energy of a single lattice point
156             *
157             * @param  x  Coordinate of Lattice Point
158             * @param  y  Coordinate of Lattice Point
159             * @return    The energy of the point at (X,Y)
160             */
161            public int calculateEnergy(int x, int y) {
162                    int[] nearestNeighboursArray = new int[4];
163                    int targetCellState = latticeArray[x][y];
164                    int energy = 0;
165    
166                    //Periodic Boundary Conditions
167                    if (y == 0) {
168                            nearestNeighboursArray[0] = latticeArray[x][size - 1];
169                    } else {
170                            nearestNeighboursArray[0] = latticeArray[x][y - 1];
171                    }
172    
173                    if (x == 0) {
174                            nearestNeighboursArray[1] = latticeArray[size - 1][y];
175                    } else {
176                            nearestNeighboursArray[1] = latticeArray[x - 1][y];
177                    }
178    
179                    if (y == (size - 1)) {
180                            nearestNeighboursArray[2] = latticeArray[x][0];
181                    } else {
182                            nearestNeighboursArray[2] = latticeArray[x][y + 1];
183                    }
184    
185                    if (x == (size - 1)) {
186                            nearestNeighboursArray[3] = latticeArray[0][y];
187                    } else {
188                            nearestNeighboursArray[3] = latticeArray[x + 1][y];
189                    }
190    
191                    // This does the work of the Krönecker ð-function
192                    for (int i = 0; i < 4; i++) {
193                            if (nearestNeighboursArray[i] == targetCellState) {
194                                    energy++;
195                            }
196                    }
197    
198                    return (-1 * energy);
199            }

203            /**
204             *  Run the Metropolis Algorithm
205             *
206             * @return    The Lattice point which was selected, for Drawing the Canvas
207             */
208            public LatticePoint runMetropolis() {
209                    int oldEnergy;
210                    int newEnergy;
211                    int energyDifference;
212                    int systemEnergyDifference;
213                    LatticePoint chosen;
214    
215                    // Pick a random lattice point to Flip
216                    int xToFlip = (int) (Math.random() * size);
217                    int yToFlip = (int) (Math.random() * size);
218    
219                    // Remember the old state in case we are gonna flip back
220                    int oldState = latticeArray[xToFlip][yToFlip];
221    
222                    // Pick a different random state to flip it to
223                    int flipToState = (int) (Math.random() * states) + 1;
224                    while (flipToState == oldState) {
225                            flipToState = (int) (Math.random() * states) + 1;
226                    }
227    
228                    chosen = new LatticePoint(xToFlip, yToFlip, flipToState, true);
229    
230                    // Calculate old state Energy
231                    oldEnergy = calculateEnergy(xToFlip, yToFlip);
232    
233                    // Flip the state
234                    latticeArray[xToFlip][yToFlip] = flipToState;
235    
236                    // Calculate the Energy of the new flipped state
237                    newEnergy = calculateEnergy(xToFlip, yToFlip);
238    
239                    // This variable can take on values from -noOfNeighbours to noOfNeighbours
240                    energyDifference = newEnergy - oldEnergy;
241    
242                    // The difference in energy of the system is twice the flipped cell difference
243                    systemEnergyDifference = 2 * energyDifference;
244    
245                    // If the system is more stable (lower energy) continue
246                    if (energyDifference < 1) {
247                            systemEnergy += systemEnergyDifference;
248    
249                            if (oldState <= states) {
250                                    magnetisation[oldState - 1]--;
251                            }
252                            magnetisation[flipToState - 1]++;
253                            return chosen;
254                    } else {
255                            // Otherwise flip with a probability based on temperature
256                            double flipProb = Math.random();
257                            if (tempGradient != 0) {
258                                    calculateExponentials();
259                            }
260    
261                            if (exponentials[energyDifference - 1] > flipProb) {
262                                    systemEnergy += systemEnergyDifference;
263                                    if (oldState <= states) {
264                                            magnetisation[oldState - 1]--;
265                                    }
266                                    magnetisation[flipToState - 1]++;
267                                    return chosen;
268                            } else {
269                                    latticeArray[xToFlip][yToFlip] = oldState;
270                                    chosen.setState(oldState);
271                                    chosen.setChanged(false);
272                                    return chosen;
273                            }
274                    }
275            }

278            /**
279             *  Run the Wolff Algorithm
280             *
281             * @return    The Cluster which was flipped, for drawing the canvas
282             */
283            public LatticeCluster runWolff() {
284                    addProb = (1.0 - Math.exp(-1.0 / temp));
285    
286                    resetClusterArray();
287    
288                    int xToFlip = (int) (Math.random() * size);
289                    int yToFlip = (int) (Math.random() * size);
290    
291                    int oldState = latticeArray[xToFlip][yToFlip];
292    
293                    int newState = (int) (Math.random() * states) + 1;
294                    while (newState == oldState) {
295                            newState = (int) (Math.random() * states) + 1;
296                    }
297    
298                    growCluster(xToFlip, yToFlip, oldState, newState);
299                    LatticeCluster safe = new LatticeCluster(oldState, newState, cluster);
300                    return safe;
301            }

304            /**
305             *  Flip the selected point and try to add its neighbours to the cluster
306             *
307             * @param  x         Coordinate of the point to flip
308             * @param  y         Coordinate of the point to flip
309             * @param  oldState  The initial state of the point to flip
310             * @param  newState  The new state of the point to flip
311             */
312            private void growCluster(int x, int y, int oldState, int newState) {
313                    cluster[x][y] = true;
314    
315                    int oldEnergy = calculateEnergy(x, y);
316    
317                    latticeArray[x][y] = newState;
318    
319                    int newEnergy = calculateEnergy(x, y);
320    
321                    int energyDiff = newEnergy - oldEnergy;
322                    systemEnergy += 2 * energyDiff;
323    
324                    int xPrev = (x == 0) ? size - 1 : x - 1;
325                    int xNext = (x == size - 1) ? 0 : x + 1;
326                    int yPrev = (y == 0) ? size - 1 : y - 1;
327                    int yNext = (y == size - 1) ? 0 : y + 1;
328    
329                    if (!cluster[xPrev][y]) {
330                            tryAdd(xPrev, y, oldState, newState);
331                    }
332                    if (!cluster[xNext][y]) {
333                            tryAdd(xNext, y, oldState, newState);
334                    }
335                    if (!cluster[x][yPrev]) {
336                            tryAdd(x, yPrev, oldState, newState);
337                    }
338                    if (!cluster[x][yNext]) {
339                            tryAdd(x, yNext, oldState, newState);
340                    }
341            }
342    
343    
344            /**
345             *  Try to add a Lattice point to the cluster
346             *
347             * @param  x         Coordinate of the point to flip
348             * @param  y         Coordinate of the point to flip
349             * @param  oldState  The initial state of the point to flip
350             * @param  newState  The new state of the point to flip
351             */
352            private void tryAdd(int x, int y, int oldState, int newState) {
353                    if (latticeArray[x][y] == oldState) {
354                            if (Math.random() < addProb) {
355                                    growCluster(x, y, oldState, newState);
356                            }
357                    }
358            }

396            /**
397             *  Change the size of the lattice while the simulation is running
398             *  If the size if decreasing, discard the hidden points
399             *  If the size is increasing, randomise the extra points
400             */
401            public void changeLatticeSize() {
402                    int[][] newLattice = new int[newSize][newSize];
403                    cluster = new boolean[newSize][newSize];
404    
405                    if (newSize < size) {
406                            for (int i = 0; i < newSize; i++) {
407                                    for (int j = 0; j < newSize; j++) {
408                                            newLattice[i][j] = latticeArray[i][j];
409                                    }
410                            }
411                            latticeArray = newLattice;
412                            size = newSize;
413                            systemEnergy = calculateTotalSystemEnergy();
414                    } else {
415                            int[][] oldLattice = latticeArray;
416                            latticeArray = newLattice;
417                            int oldSize = size;
418                            size = newSize;
419                            randomiseLattice();
420                            for (int i = 0; i < oldSize; i++) {
421                                    for (int j = 0; j < oldSize; j++) {
422                                            latticeArray[i][j] = oldLattice[i][j];
423                                    }
424                            }
425                            systemEnergy = calculateTotalSystemEnergy();
426                    }
427            }
428

430            /**
431             *  Change the number of states in the lattice and recalculate the no of points in each state
432             */
433            public void changeNoOfStates() {
434                    int[] newMagArray = new int[newStates];
435                    magnetisation = newMagArray;
436                    states = newStates;
437                    calculateSystemMagnetisation();
438    
439            }

595    class LatticeFrame extends Frame implements ActionListener {
596    
597    
598            Lattice richie;
599            LatticeCanvas pottsDisplay;
600            LatticeControlPanel controls;
601    
602    
603            /**
604             *Constructor for the LatticeFrame object
605             */
606            public LatticeFrame() {
607                    setTitle("Emma's Lattice Simulator");
608                    this.controls = new LatticeControlPanel();
609    
610                    int size = Integer.parseInt(controls.size.getText());
611                    int states = Integer.parseInt(controls.states.getText());
612                    double temp = Double.parseDouble(controls.temp.getText());
613                    double tempGradient = Double.parseDouble(controls.tempGradient.getText());
614    
615                    this.richie = new Lattice(size, states, temp, tempGradient);
616                    this.pottsDisplay = new LatticeCanvas(richie);
617    
618                    this.add(pottsDisplay, BorderLayout.CENTER);
619                    this.add(controls, BorderLayout.SOUTH);
620    
621                    // add action listerners so the program 'listens' to the buttons
622                    controls.reset.addActionListener(this);
623                    controls.startPause.addActionListener(this);
624                    controls.algorithm.addActionListener(this);
625                    controls.update.addActionListener(this);
626    
627                    // code to close the simulation if the close button is pushed
628                    addWindowListener(
629                            new WindowAdapter() {
630                                    public void windowClosing(WindowEvent e) {
631                                            System.exit(0);
632                                    }
633                            });
634            }

636    
637            /**
638             *  Listen for user input
639             *
640             * @param  e  The user event
641             */
642            public void actionPerformed(ActionEvent e) {
643                    try {
644                            // If the update button is pushed
645                            if (e.getSource() == controls.update) {
646    
647                                    int size;
648    
649                                    int states;
650    
651                                    // Get the lattice parameters from the control panel
652                                    states = Integer.parseInt(controls.states.getText());
653                                    size = Integer.parseInt(controls.size.getText());
654    
655                                    // If the size has changed, update the lattice object
656                                    if (richie.size != size) {
657                                            richie.newSize = size;
658                                            richie.sizeChanged = true;
659                                            controls.size.setText(String.valueOf(size));
660                                    }
661    
662                                    // If the number of states has changed update the lattice
663                                    if (richie.states != states) {
664                                            richie.newStates = states;
665                                            richie.statesChanged = true;
666                                            controls.states.setText(String.valueOf(states));
667                                    }
668    
669                                    richie.temp = Double.parseDouble(controls.temp.getText());
670                                    richie.tempGradient = Double.parseDouble(controls.tempGradient.getText());
671                                    richie.calculateExponentials();
672                            }
673    
674                            // The reset button doubles up as the update button when the simulation is paused
675                            // The update allows the user to watch each algorithm, step by step
676                            if (e.getSource() == controls.reset) {
677                                    if (richie.running == true) {
678                                            richie.resetMe = true;
679                                    } else {
680                                            if (richie.metropolis == true) {
681                                                    LatticePoint drawMe = richie.runMetropolis();
682                                                    if (drawMe.hasChanged()) {
683                                                            pottsDisplay.drawPoint(drawMe);
684                                                    }
685                                            } else {
686                                                    LatticeCluster drawMe = richie.runWolff();
687                                                    pottsDisplay.drawCluster(drawMe);
688                                            }
689                                    }
690                            }
691    
692                            if (e.getSource() == controls.startPause) {
693                                    if (richie.running == false) {
694                                            controls.startPause.setLabel("Pause");
695                                            controls.reset.setLabel("Reset");
696                                            richie.running = true;
697    
698                                    } else if (richie.running == true) {
699                                            controls.startPause.setLabel("Continue");
700                                            controls.reset.setLabel("Step");
701                                            richie.running = false;
702                                    }
703                            }
704                            if (e.getSource() == controls.algorithm) {
705                                    if (richie.metropolis == true) {
706                                            richie.metropolis = false;
707                                            controls.temp.setText(String.valueOf(richie.temp));
708                                            controls.algorithm.setLabel("Metropolis");
709                                    } else {
710                                            richie.metropolis = true;
711                                            controls.temp.setText(String.valueOf(richie.temp));
712                                            controls.algorithm.setLabel("Wolff");
713    
714                                    }
715                            }
716                    } catch (Exception ex) {
717                            System.out.println("Error in the Input");
718                            System.exit(1);
719                    }
720            }

723            /**
724             *  Run the Simulation
725             */
726            public void runSimulation() {
727    
728                    int i = 0;
729                    while (true) {
730    
731                            // Check all of the 'Changed' flags every iteration and if any are true, update the lattice
732                            if (richie.running == true) {
733                                    if (richie.sizeChanged == true) {
734                                            richie.changeLatticeSize();
735                                            richie.sizeChanged = false;
736                                            pottsDisplay.repaint();
737                                    }
738    
739                                    if (richie.statesChanged == true) {
740                                            richie.changeNoOfStates();
741                                            richie.statesChanged = false;
742                                    }
743    
744                                    if (richie.resetMe == true) {
745                                            richie.resetSimulation();
746                                            pottsDisplay.repaint();
747                                            richie.resetMe = false;
748                                    }
749    
750                                    if (richie.tempGradient != 0) {
751                                            if ((richie.temp + richie.tempGradient) > 0) {
752                                                    richie.temp += richie.tempGradient;
753                                            } else {
754                                                    richie.temp = 0;
755                                            }
756                                            // Update the temperature on the control panel every 10000 iterations
757                                            if (i == 10000) {
758                                                    controls.temp.setText(String.valueOf(richie.temp));
759                                                    i = 0;
760                                            }
761                                            i++;
762                                    }
763    
764                                    if (richie.metropolis == true) {
765                                            LatticePoint drawMe = richie.runMetropolis();
766                                            if (drawMe.hasChanged()) {
767                                                    pottsDisplay.drawPoint(drawMe);
768                                            }
769                                    } else {
770                                            LatticeCluster drawMe = richie.runWolff();
771                                            pottsDisplay.drawCluster(drawMe);
772                                    }
773                            }
774                    }
775            }

779     *  The Canvas where the Lattice is Drawn
780     *
781     * @author    Richard Hanes
782     */
783    class LatticeCanvas extends Canvas {
784    
785    
786            Lattice richie;
787            int size, states, w, h, x, y;
788            Graphics safe;
789    
790    
791            /**
792             * Constructor for the LatticeCanvas object
793             *
794             * @param  richie  The lattice to be drawn
795             */
796            public LatticeCanvas(Lattice richie) {
797                    // set the background colour of the Canvas
798                    setBackground(Color.black);
799                    this.richie = richie;
800            }
801    
802    
803            /**
804             *  Allows for faster repainting
805             *
806             * @param  g  Graphics object of the Canvas
807             */
808            public void update(Graphics g) {
809                    paint(g);
810            }
811    
812    
813            /**
814             *  Draw an individual point of the lattice
815             *
816             * @param  drawMe  The point to be drawn
817             */
818            public void drawPoint(LatticePoint drawMe) {
819                    safe = getGraphics();
820                    Dimension d = getSize();
821                    int x;
822                    int y;
823                    int w;
824                    int h;
825                    int state;
826    
827                    size = richie.getSize();
828                    states = richie.getStates();
829    
830                    // determine size of each spin particle
831                    x = drawMe.getX();
832                    y = drawMe.getY();
833                    state = drawMe.getState();
834                    w = d.width / size;
835                    h = d.height / size;
836    
837                    // paint the lattice
838                    x = w * x;
839                    y = h * y;
840                    safe.setColor(colour(state, states));
841                    safe.fillOval(x, y, w, h);
842    
843            }
844    
845    
846            /**
847             *  Draw a cluster of Points
848             *
849             * @param  drawMe  The cluster of points to be drawn
850             */
851            public void drawCluster(LatticeCluster drawMe) {
852                    size = richie.getSize();
853                    for (int i = 0; i < size; i++) {
854                            for (int j = 0; j < size; j++) {
855                                    if (drawMe.isInCluster(i, j)) {
856                                            LatticePoint singleSpin = new LatticePoint(i, j, drawMe.getNewState(), true);
857                                            drawPoint(singleSpin);
858                                    }
859                            }
860                    }
861            }
862    
863    
864            /**
865             *  Draw the entire Lattice
866             *
867             * @param  g  Graphics object of the Canvas
868             */
869            public void paint(Graphics g) {
870                    // get current size of Gui
871                    Dimension d = getSize();
872                    g.setColor(Color.black);
873    
874                    g.fillRect(0, 0, d.width, d.height);
875    
876                    size = richie.getSize();
877                    states = richie.getStates();
878    
879                    // determine size of each spin particle
880                    w = d.width / size;
881                    h = d.height / size;
882    
883                    // paint the lattice
884                    for (int i = 0; i < size; i++) {
885                            for (int j = 0; j < size; j++) {
886                                    x = w * i;
887                                    y = h * j;
888                                    g.setColor(colour(richie.latticeArray[i][j], states));
889                                    g.fillOval(x, y, w, h);
890                            }
891                    }
892            }
893    
894    
895            /**
896             *  This method uses the rgb color model
897             *  to set the colour depending on ratio
898             *  of currentState to totalStates.
899             *
900             * @param  i       The state of the current point
901             * @param  states  The total number of states
902             * @return         A color object
903             */
904            private Color colour(int i, int states) {
905    
906                    float r;
907                    float g;
908                    float b;
909                    Color spinColor;
910    
911                    // Vary the colours as the number of states increases
912                    switch (i) {
913                                    case 7:
914                                            spinColor = Color.yellow;
915                                            break;
916                                    case 8:
917                                            spinColor = Color.blue;
918                                            break;
919                                    case 9:
920                                            spinColor = Color.orange;
921                                            break;
922                                    case 10:
923                                            spinColor = Color.pink;
924                                            break;
925                                    default:
926    
927                                            g = (float) i / (float) states;
928                                            r = 1.0f - g;
929                                            b = 0.0f;
930                                            spinColor = new Color(r, g, b);
931                                            break;
932                    }
933                    return spinColor;
934            }
935    }
